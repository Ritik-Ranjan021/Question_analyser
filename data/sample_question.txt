    int i, j;
    int v, x;
    if (n <= m) return;

    while(1)
    {
        do i = i + 1; while (a[i] < v);
        do j = j − 1; while (a[j] > v);

        x = a[i]; a[i] = a[j]; a[j] = x;
    }
    x = a[i]; a[i] = a[n]; a[n] = x;
    quicksort(m, j);

}
